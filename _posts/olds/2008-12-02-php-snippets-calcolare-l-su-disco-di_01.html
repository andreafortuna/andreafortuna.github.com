function diskspace($dir)
{
   $s = stat($dir);
   $space = $s["size"];
   if (is_dir($dir))
   {
     $dh = opendir($dir);
     while (($file = readdir($dh)) !== false)
       if ($file != "." and $file != "..")
         $space += dskspace($dir."/".$file);
     closedir($dh);
   }
   return $space;
}

restituisce le dimensioni in bytes della directory specificata, comprese eventuali sotto-directory.
